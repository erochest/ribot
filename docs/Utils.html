This splits a list (string) on a character.
             		                            
split :: Eq a => a -> [a] -> [[a]]
split _  []   = []
split on list = case rest of
                    (on:rest') -> prefix : split on rest'
                    []         -> [prefix]
    where (prefix, rest) = span (/=on) list
